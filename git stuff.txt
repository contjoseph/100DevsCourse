git init

git status   //shows the staging area

git add <filename>  ///it adds it to staging

git commit -m "My message about the commit"

git log --oneline   //shows all of you 

git add -A   //adds all files that aren't currently tracked

git commit --amend --no-edit //amends/modifies your last commit 
                             //without the needs to add a new message

git add .   // with a period adds all of the files that needs to be tracked
